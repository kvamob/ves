            // Откроем поток

            FileStream vesFile = new FileStream("file.ves", FileMode.Open, FileAccess.Read);

            fileType = GetFileType(inputFile);		// Тип ВЭЗ-файла


            // Загрузить заголовок

            hdrVES.Load(vesFile, fileType);
